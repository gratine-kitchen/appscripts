const SOURCE_SS_ID = '1Gjj6AdnMuZRw0_48qI0EbZWPk3JbBYlRk1vrz6HBcVU';
const RECEIPT_SS_ID = '1JhTvCsAlABfwmPcE4P3rrZZt29aRYQIGGclvVrTK2bY';
const TEMPLATE_SHEET_NAME = 'ReceiptTemplate';
const SOURCE_SHEETS = {
  bookings: 'Bookings',
  menuForBooking: 'MenuForBooking',
  menuItems: 'MenuItems',
  guests: 'Guests'
};

const TEMPLATE = {
  firstItemRow: 13,
  lastItemRow: 17,
  subtotalRow: 18,
  serviceChargeRow: 19,
  corkageRow: 20,
  discountRow: 21,
  totalRow: 22
};

function promptGenerateReceipt() {
  const ui = SpreadsheetApp.getUi();
  const response = ui.prompt('Generate Receipt', 
    'Enter booking ItemId:', 
    ui.ButtonSet.OK_CANCEL);
  
  if (response.getSelectedButton() !== ui.Button.OK) return;
  
  const bookingId = String(response.getResponseText()).trim();
  if (!bookingId) {
    ui.alert('No booking ID entered.');
    return;
  }
  
  try {
    const sheetName = generateReceipt(bookingId);
    ui.alert('Success', `Receipt created: ${sheetName}`, ui.ButtonSet.OK);
  } catch (e) {
    ui.alert('Error', e.toString(), ui.ButtonSet.OK);
  }
}

function generateReceipt(bookingId) {

  const receiptSS = SpreadsheetApp.openById(RECEIPT_SS_ID);
  const sourceSS = SpreadsheetApp.openById(SOURCE_SS_ID);

  // Get data tables
  const bookings = getTable(sourceSS, SOURCE_SHEETS.bookings);
  const menuForBooking = getTable(sourceSS, SOURCE_SHEETS.menuForBooking);
  const menuItems = getTable(sourceSS, SOURCE_SHEETS.menuItems);
  const guests = getTable(sourceSS, SOURCE_SHEETS.guests);

  // Find booking
  const booking = findRow(bookings.rows, bookings.headers, 'ItemId', bookingId);
  if (!booking) throw new Error(`Booking not found: ${bookingId}`);

  // Find guest
  const guestId = booking[bookings.headers.indexOf('PrimaryGuest')];
  const guest = findRow(guests.rows, guests.headers, 'ItemId', guestId);
  const guestName = guest ? String(guest[guests.headers.indexOf('GuestName')]) : 'Unknown Guest';
  
  const bookingDate = booking[bookings.headers.indexOf('Date')];
  const serviceChargePct = parseFloat(booking[bookings.headers.indexOf('ServiceCharge')] || 0);
  const discountPct = parseFloat(booking[bookings.headers.indexOf('Discount')] || 0);
  const depositPaid = parseFloat(booking[bookings.headers.indexOf('DepositPaid')] || 0);

  // Create sheet
  const sheetName = `${formatDateForName(bookingDate)} (${guestName})`;
  deleteExistingSheet(receiptSS, sheetName);

  const templateSheet = receiptSS.getSheetByName(TEMPLATE_SHEET_NAME);
  if (!templateSheet) throw new Error('ReceiptTemplate sheet not found');
  
  const newSheet = templateSheet.copyTo(receiptSS).setName(sheetName);
  receiptSS.setActiveSheet(newSheet);
  receiptSS.moveActiveSheet(1);

  // Fill static fields
  newSheet.getRange('B8').setValue(formatDateForDisplay(bookingDate));
  newSheet.getRange('B9').setValue(guestName);

  // Get line items
  const lineItems = getLineItems(menuForBooking.rows, menuForBooking.headers, 
                                menuItems.rows, menuItems.headers, 
                                bookingId);

  // Insert rows if needed
  const neededRows = lineItems.length;
  const templateRows = TEMPLATE.lastItemRow - TEMPLATE.firstItemRow + 1;
  const numRowsDelta = neededRows - templateRows;
  if (numRowsDelta > 0) {
   newSheet.insertRowsBefore(TEMPLATE.subtotalRow, numRowsDelta);
    copyTemplateRowStyle(newSheet, TEMPLATE.lastItemRow, numRowsDelta);
  } else if (numRowsDelta < 0) {
    const unneededRows = Math.abs(numRowsDelta);
    newSheet.deleteRows(TEMPLATE.subtotalRow - unneededRows, unneededRows);
  }

  // Fill line items
  fillLineItems(newSheet, lineItems, TEMPLATE.firstItemRow);

  // Recalculate totals (preserve template formulas where possible)
  updateTotals(newSheet, TEMPLATE.firstItemRow, lineItems.length, numRowsDelta,
               serviceChargePct, discountPct, depositPaid);

  return sheetName;
}

function getTable(ss, sheetName) {
  const sheet = ss.getSheetByName(sheetName);
  if (!sheet) throw new Error(`Sheet not found: ${sheetName}`);

  const values = sheet.getDataRange().getValues();
  const headers = values[0].map(String).map(h => h.trim());
  const rows = values.slice(1);
  return { headers, rows };
}

function findRow(rows, headers, keyCol, keyValue) {
  const col = headers.indexOf(keyCol);
  return rows.find(row => String(row[col]).trim() === String(keyValue).trim());
}

function getLineItems(mbRows, mbHeaders, miRows, miHeaders, bookingId) {
  const bookingCol = mbHeaders.indexOf('Booking');
  const dishCol = mbHeaders.indexOf('Dish');
  const qtyCol = mbHeaders.indexOf('Qty');
  const orderCol = mbHeaders.indexOf('Order');
  const upgradeWaivedCol = mbHeaders.indexOf('UpgradeWaived');
  const complimentaryCol = mbHeaders.indexOf('Complimentary');

  return mbRows
    .filter(row => String(row[bookingCol]).trim() === String(bookingId).trim())
    .filter(row => !truthy(row[upgradeWaivedCol]))
    .sort((a, b) => {
    return Number(a[orderCol]) - Number(b[orderCol]);
  })
    .map(row => {
      const dishId = row[dishCol];
      const mi = findRow(miRows, miHeaders, 'Dish', dishId);
      if (!mi) return null;

      const upgradePrice = parseFloat(mi[miHeaders.indexOf('UpgradePrice')] || 0);
      if (upgradePrice <= 0 && !truthy(row[complimentaryCol])) return null;

      return {
        dish: dishId,
        qty: parseFloat(row[qtyCol] || 0),
        upgradePrice: upgradePrice,
        complimentary: truthy(row[complimentaryCol])
      };
    })
    .filter(item => item !== null);
}

function fillLineItems(sheet, items, startRow) {
  items.forEach((item, i) => {
    const row = startRow + i;
    sheet.getRange(row, 1).setValue(item.dish);
    sheet.getRange(row, 2).setValue(item.qty);
    sheet.getRange(row, 3).setValue(item.upgradePrice);
    
    if (item.complimentary) {
      sheet.getRange(row, 4).setValue('complimentary').setHorizontalAlignment('right');
    } else {
      // Copy formula from first row if available
      const firstFormula = sheet.getRange(startRow, 4).getFormulaR1C1();
      if (firstFormula) {
        sheet.getRange(row, 4).setFormulaR1C1(firstFormula);
      }
    }
  });
}

function copyTemplateRowStyle(sheet, sourceRow, numRows) {
  const sourceRange = sheet.getRange(sourceRow, 1, 1, sheet.getLastColumn());
  for (let i = 0; i < numRows; i++) {
    const targetRow = sourceRow + i + 1;
    sourceRange.copyTo(sheet.getRange(targetRow, 1, 1, sheet.getLastColumn()), 
                      SpreadsheetApp.CopyPasteType.PASTE_FORMAT, false);
  }
}

function updateTotals(sheet, firstRow, numRows, numRowsDelta, servicePct, discountPct, depositPaid) {
  const subtotalRow = TEMPLATE.subtotalRow + numRowsDelta;
  const serviceRow = TEMPLATE.serviceChargeRow + numRowsDelta;
  const discountRow = TEMPLATE.discountRow + numRowsDelta;
  const totalRow = TEMPLATE.totalRow + numRowsDelta;

  const subtotalRange = `D${firstRow}:D${firstRow + numRows - 1}`;
  
  sheet.getRange(subtotalRow, 4).setFormula(`=SUM(D${firstRow}:D${firstRow + numRows - 1})`);
  
  if (servicePct > 0)
  {
    sheet.getRange(serviceRow, 4).setFormula(`=D${subtotalRow}*${servicePct}`);
  }
  else
  {
    sheet.getRange(serviceRow, 4).setValue("complimentary");
  }
  
  if (discountPct > 0)
  {
    sheet.getRange(discountRow, 3).setValue('Discount (-' + (discountPct * 100) + '%)');
    sheet.getRange(discountRow, 4).setFormula(`=D${subtotalRow}*-${discountPct}`);
  }
  else
  {
    sheet.hideRows(discountRow);
  }

  let totalFormula = `=SUM(D${subtotalRow}:D${totalRow - 1})`;
  
  if (depositPaid > 0) {
    const depositRow = totalRow - 1;
    sheet.insertRowBefore(totalRow);
    sheet.getRange(depositRow, 4).setValue(-depositPaid)
      .setNumberFormat('#,##0.00;(#,##0.00)')
      .setHorizontalAlignment('right');
    totalFormula = `=D${subtotalRow}+D${serviceRow}-D${subtotalRow}*${discountPct}+D${depositRow}`;
  }
  
  sheet.getRange(totalRow, 4).setFormula(totalFormula);
}

function deleteExistingSheet(ss, name) {
  const sheet = ss.getSheetByName(name);
  if (sheet) ss.deleteSheet(sheet);
}

function truthy(value) {
  if (value === null || value === undefined) return false;
  if (typeof value === 'boolean') return value;
  const str = String(value).toLowerCase().trim();
  return str === 'true' || str === '1' || str === 'yes';
}

function formatDateForDisplay(date) {
  const d = new Date(date);
  return Utilities.formatDate(d, Session.getScriptTimeZone(), 'MM/dd/yyyy');
}

function formatDateForName(date) {
  const d = new Date(date);
  return Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyyMMdd');
}

 function copySheetsToNewWorkbook() {
  // Source spreadsheet and sheet containing the list of sheet names.
  var sourceSpreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  var sheetWithNames = sourceSpreadsheet.getSheetByName("Sheet4");

  // Get the list of sheet names from the specified sheet.
  var sheetNamesRange = sheetWithNames.getDataRange();
  var sheetNamesValues = sheetNamesRange.getValues();

  // Create a new spreadsheet.
  var newSpreadsheet = SpreadsheetApp.create("Gratine Receipts for 2024-25 tax filing (final)");

  // Loop through the sheet names and copy the corresponding sheets to the new spreadsheet.
  for (var i = 0; i < sheetNamesValues.length; i++) {
    var sheetName = sheetNamesValues[i][0]; // Assuming sheet names are in the first column

    // Get the sheet from the source spreadsheet.
    var sheetToCopy = sourceSpreadsheet.getSheetByName(sheetName);

    // Check if the sheet exists in the source spreadsheet.
    if (sheetToCopy) {
      // Copy the sheet to the new spreadsheet.
      sheetToCopy.copyTo(newSpreadsheet);
    } else {
      Logger.log("Sheet '" + sheetName + "' not found in the source spreadsheet.");
    }
  }

  // Remove the default "Sheet1" from the new spreadsheet.
  newSpreadsheet.deleteSheet(newSpreadsheet.getSheetByName("Sheet1"));

  Logger.log("Sheets copied to new spreadsheet: " + newSpreadsheet.getUrl());
}

function doPost(e) {
  try {
    const body = e && e.postData && e.postData.contents ? JSON.parse(e.postData.contents) : {};
    const bookingId = body.bookingId;

    if (!bookingId) {
      return ContentService
        .createTextOutput(JSON.stringify({ error: 'Missing bookingId' }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    const result = generateReceipt(bookingId);
    
    return ContentService
      .createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ error: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
